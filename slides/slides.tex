\section{Problema e dados}
%=================================================
\sectionframe

\begin{frame}{Motiva\c{c}\~ao}
    \begin{block}{Pergunta central}
        Quais perfis epidemiol\'ogicos emergem ao combinar incid\^encia, tend\^encia temporal e contexto municipal, e como munic\'ipios transitam entre esses perfis ao longo das semanas?
    \end{block}

    \begin{itemize}
        \item Dengue varia no \textbf{territ\'orio} e no \textbf{tempo}: totais anuais escondem picos semanais.
        \item Olhar s\'o casos absolutos favorece munic\'ipios populosos.
        \item O objetivo \'e descobrir \textbf{est\'agios de propaga\c{c}\~ao} recorrentes entre munic\'ipios e semanas.
        \item Os perfis emergem do agrupamento a partir dos padr\~oes no dado semanal.
    \end{itemize}
\end{frame}

\begin{frame}{Base de dados}
    \begin{columns}[T]
        \begin{column}{0.50\textwidth}
            \textbf{Fonte principal}
            \begin{itemize}
                \item SINAN / OpenDataSUS
                \item Notifica\c{c}\~oes de arboviroses (dengue)
                \item CSVs nacionais: \texttt{DENGBR16.csv} a \texttt{DENGBR26.csv}
            \end{itemize}
        \end{column}
        \begin{column}{0.45\textwidth}
            \textbf{Recorte da an\'alise principal}
            \begin{itemize}
                \item Bahia (\uf{BA}), ano 2024
                \item 352 munic\'ipios eleg\'iveis ($\geq 30$ casos/ano)
                \item 18.304 linhas: munic\'ipio $\times$ semana epidemiol\'ogica
            \end{itemize}
        \end{column}
    \end{columns}

    \vspace{0.5em}
    \begin{block}{Campos e refer\^encias auxiliares}
        \texttt{ID\_MUNICIP}, \texttt{SEM\_NOT} no SINAN; popula\c{c}\~ao e \'{a}rea municipal (IBGE) para incid\^encia e densidade.
    \end{block}

    \smallnote{Escolha de $K$ foi calibrada no Nordeste inteiro (9 UFs); a narrativa de features usa BA como caso principal.}
\end{frame}

\begin{frame}{Unidade anal\'itica}
    \begin{center}
    \begin{tikzpicture}[node distance=1.4cm, every node/.style={font=\small}]
        \tikzset{flowstep/.style={rectangle, rounded corners, draw=black!60, fill=MedianUltraLightOrange, minimum width=2.4cm, minimum height=0.85cm, align=center}}
        \node[flowstep] (sinan) {Notifica\c{c}\~oes\\SINAN};
        \node[flowstep, right of=sinan, xshift=1.6cm] (panel) {Painel\\munic\'ipio/semana};
        \node[flowstep, right of=panel, xshift=1.6cm] (feat) {Features\\versionadas};
        \node[flowstep, right of=feat, xshift=1.6cm] (km) {K-means\\$K=4$};
        \node[flowstep, right of=km, xshift=1.6cm] (trans) {Transi\c{c}\~oes\\$t \rightarrow t{+}1$};
        \draw[->, thick] (sinan) -- (panel);
        \draw[->, thick] (panel) -- (feat);
        \draw[->, thick] (feat) -- (km);
        \draw[->, thick] (km) -- (trans);
    \end{tikzpicture}
    \end{center}

    \begin{itemize}
        \item Cada linha = \textbf{situa\c{c}\~ao epidemiol\'ogica} de um munic\'ipio em uma semana.
        \item Features padronizadas antes do K-means
        \item Clusters reordenados pela m\'edia de casos
    \end{itemize}
\end{frame}

\begin{frame}{Por que aprendizado n\~ao supervisionado?}
    \begin{itemize}
        \item O dado bruto registra casos por munic\'ipio/semana, sem r\'otulos de alerta ou surto.
        \item Queremos \textbf{descobrir} padr\~oes recorrentes e depois interpret\'a-los como est\'agios.
        \item A valida\c{c}\~ao posterior (ex.: alerta $\rightarrow$ alta incid\^encia na semana seguinte) checa a utilidade dos perfis encontrados.
    \end{itemize}

    \begin{block}{Formula\c{c}\~ao segura}
        Ferramenta explorat\'oria para vigil\^encia: identificar perfis e acompanhar transi\c{c}\~oes territoriais.
    \end{block}
\end{frame}

\begin{frame}{v0 vs v1: por que normalizar?}
    \small
    \begin{columns}[T]
        \begin{column}{0.48\textwidth}
            \textbf{v0 (casos absolutos)}
            \begin{itemize}
                \item Silhouette alta (0{,}92), mas perfil trivial.
                \item Cluster 3: m\'edia de 2.540 casos/semana (Salvador e grandes centros).
                \item 97\% das linhas no cluster de baixa contagem absoluta.
            \end{itemize}
        \end{column}
        \begin{column}{0.48\textwidth}
            \textbf{v1 (+ incid\^encia / 100 mil)}
            \begin{itemize}
                \item Mesmo munic\'ipio grande pode ter incid\^encia moderada.
                \item Interior com poucos casos deixa de ser ``igual'' a capital em repouso.
                \item Base para comparar \textbf{munic\'ipios entre si} pelo risco relativo.
            \end{itemize}
        \end{column}
    \end{columns}

    \vspace{0.5em}
    \begin{block}{Insight}
        Em uma \u{u}nica cidade, incid\^encia $\approx$ casos $\times$ constante. No painel BA, v1 passa a importar de verdade.
    \end{block}
\end{frame}

\section{Conclus\~ao}
%=================================================
\sectionframe

\begin{frame}{Limita\c{c}\~oes}
    \begin{itemize}
        \item Recorte principal: Bahia 2024; generaliza\c{c}\~ao a outros anos e UFs ainda a validar.
        \item Sazonalidade c\'iclca (sin/cos) foi testada e \textbf{descartada}: piorou silhouette sem ganho interpret\'avel.
        \item Clima defasado (chuva, temperatura) ainda n\~ao entrou no modelo.
        \item Transi\c{c}\~oes s\~ao calculadas depois do agrupamento semanal.
        \item Subnotifica\c{c}\~ao e heterogeneidade da vigil\^encia municipal limitam comparabilidade.
    \end{itemize}
\end{frame}

\begin{frame}{Pr\'oximos passos}
    \begin{itemize}
        \item Estender painel multivariado a outras UFs do Nordeste.
        \item Testar GMM para probabilidade de pertencimento a cada perfil.
        \item Integrar chuva e temperatura defasadas, se cobertura clim\'atica permitir.
        \item Validar transi\c{c}\~oes alerta $\rightarrow$ surto como checagem posterior dos perfis.
        \item Comparar clusters com faixas fixas de incid\^encia (baseline epidemiol\'ogico simples).
    \end{itemize}
\end{frame}

\begin{frame}{Conclus\~ao}
    \begin{block}{Mensagem principal}
        Clusteriza\c{c}\~ao n\~ao supervisionada transforma s\'eries semanais de dengue em \textbf{est\'agios interpret\'aveis de propaga\c{c}\~ao} e permite acompanhar \textbf{transi\c{c}\~oes} territoriais.
    \end{block}

    \begin{enumerate}
        \item v0 mostra o limite de casos absolutos; v1 a v4 adicionam din\^amica epidemiol\'ogica.
        \item v5 traz contexto urbano (densidade) com pequeno custo em m\'etrica interna.
        \item $K=4$ veio de calibra\c{c}\~ao no Nordeste e se mant\'em leg\'ivel na apresenta\c{c}\~ao.
        \item Resultado pr\'atico: ferramenta explorat\'oria para vigil\^encia epidemiol\'ogica municipal.
    \end{enumerate}
\end{frame}
